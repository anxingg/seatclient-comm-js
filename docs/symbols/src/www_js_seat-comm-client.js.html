  6 /**
  7  * @class
  8  * @description 座席通讯客户端
  9  * @param {string} server_url 服务端地址 
 10  * @param {function()} onopen 连接成功事件处理函数
 11  * @param {function()} onclose 连接失败事件处理函数
 12  * @param {function(evt)} onmessage 接收到数据的处理函数,evt.data获取收到的数据
 13  * @param {function()} onerror 连接失败事件处理函数
 14  * @return {seat_comm_client}
 15  */
 16 var seat_comm_client = function (server_url, onopen, onclose, onmessage, onerror) {
 17 
 18     var ob = new Object();
 19 
 20     //服务端地址
 21     ob.server_url = server_url;
 22 
 23     ob.socket = new WebSocket(ob.server_url);
 24 
 25     ob.socket.onopen = onopen;
 26 
 27     ob.socket.onclose = onclose;
 28 
 29     ob.socket.onmessage = onmessage;
 30 
 31     ob.socket.onerror = onerror;
 32 
 33     ob.send = function (data) {
 34         if (this.socket.readyState == WebSocket.OPEN) {
 35             this.socket.send(data);
 36         } else {
 37             var err = "socket.readyState=" + this.socket.readyState;
 38             throw err;
 39         }
 40     }
 41     ob.ping = function () {
 42         this.socket.ping();
 43     }
 44     ob.close = function () {
 45         this.socket.close();
 46     }
 47     return ob;
 48 }

372             }
373             else if (seat_client.callInfo.nCallType == CALL_TYPE_CONSTANTS.OutCall) {
374                 seat_client.onhangupReport(dataArray[1], dataArray[2], dataArray[3]);
375             }
376             if (seat_client.msiUser.msiState == MSI_STATE_CONSTANTS.SeatState_LoginIn)//置闲
377             {
378                 seat_client.sendsetmsistate(0);//闲
379             }
380             else if (seat_client.msiUser.msiState == MSI_STATE_CONSTANTS.SeatState_LoginOut)//置忙
381             {
382                 seat_client.sendcheckout(0);//签出
383             }
384             seat_client.changeCallState(CALL_INFO_STATE_CONSTANTS.CallState_End);
385             seat_client.initCallInfo();
386             console.log("hangupreport end");
387         }
388     }
389     else {
390         console.log("warnning:seat_client.hangupReport seat_client.callInfo.callId=", seat_client.callInfo.callId);
391     }

605 function GetDateDiff(sTime, eTime, diffType) {
606     //将计算间隔类性字符转换为小写
607     diffType = diffType.toLowerCase();
608     //作为除数的数字
609     var divNum = 1;
610     switch (diffType) {
611         case "second":
612             divNum = 1000;
613             break;
614         case "minute":
615             divNum = 1000 * 60;
616             break;
617         case "hour":
618             divNum = 1000 * 3600;
619             break;
620         case "day":
621             divNum = 1000 * 3600 * 24;
622             break;
623         default:
624             break;
625     }
626     return parseInt((new Date(eTime)
627         - new Date(sTime)) / parseInt(divNum));
628 }

630 seat_client.onidle = function () {
631     try{
632     if (seat_client.comm.socket.readyState == WebSocket.OPEN) {
633         seat_client.comm.ping();
634         var minu = GetDateDiff(seat_client.lastmsgtime, Date(), "minute");
635         if (minu >= 3 * COMM_MSGHEAD_CONSTANTS.IDLEMAX) {
636             seat_client.close();
637         }
638         else if (minu >= COMM_MSGHEAD_CONSTANTS.IDLEMAX) {
639             seat_client.sendheartbeat();
640         }
641     }
642     else {
643         if (seat_client.closeFlag == false) {
644             seat_client.close(true);
645         }
646         seat_client.onReconnect();  //通知上层应用正在重连
647         setTimeout(seat_client.connect(), 1000);
648     }
649     }
650     catch(ex){
651         console.log("seat_client.onidle error:"+ex.message);
652     }
653 }
