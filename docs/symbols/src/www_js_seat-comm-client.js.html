  7  * @class
  8  * @description 座席通讯客户端
  9  * @private
 10  * @param {string} server_url 服务端地址 
 11  * @param {function()} onopen 连接成功事件处理函数
 12  * @param {function()} onclose 连接失败事件处理函数
 13  * @param {function(evt)} onmessage 接收到数据的处理函数,evt.data获取收到的数据
 14  * @param {function()} onerror 连接失败事件处理函数
 15  * @return {seat_comm_client}
 16  */
 17 var seat_comm_client = function (server_url, onopen, onclose, onmessage, onerror) {
 18 
 19     var ob = new Object();
 20 
 21     //服务端地址
 22     ob.server_url = server_url;
 23 
 24     ob.socket = new WebSocket(ob.server_url);
 25 
 26     ob.socket.onopen = onopen;
 27 
 28     ob.socket.onclose = onclose;
 29 
 30     ob.socket.onmessage = onmessage;
 31 
 32     ob.socket.onerror = onerror;
 33 
 34     ob.send = function (data) {
 35         if (this.socket.readyState == WebSocket.OPEN) {
 36             this.socket.send(data);
 37         } else {
 38             var err = "socket send error:socket.readyState=" + this.socket.readyState;
 39             console.log(err);
 40             ob.socket.onerror(new Error(err));
 41             //throw err;
 42         }
 43     }
 44     ob.ping = function () {
 45         this.socket.ping();
 46     }
 47     ob.close = function () {
 48         this.socket.close();
 49     }
 50     return ob;
 51 }

937 function GetDateDiff(sTime, eTime, diffType) {
938     //将计算间隔类性字符转换为小写
939     diffType = diffType.toLowerCase();
940     //作为除数的数字
941     var divNum = 1;
942     switch (diffType) {
943         case "second":
944             divNum = 1000;
945             break;
946         case "minute":
947             divNum = 1000 * 60;
948             break;
949         case "hour":
950             divNum = 1000 * 3600;
951             break;
952         case "day":
953             divNum = 1000 * 3600 * 24;
954             break;
955         default:
956             break;
957     }
958     return parseInt((new Date(eTime)
959         - new Date(sTime)) / parseInt(divNum));
960 }
